export function checkSmritiReportPermissions(req: express.Request, action: string, reportId?: string): boolean {
  if (!hasPermission(req, "reports.view")) return false;
  
  if (!hasPermission(req, "settings.manage")) {
    if (action === "export" || action === "schedule" || action === "email") return false;
    if (reportId && (reportId.startsWith("RPT-FIN") || reportId === "RPT-CST-LOG")) return false;
  }
  
  if (!hasPermission(req, "roles.manage")) {
    if (reportId === "RPT-FIN-003" && (action === "export" || action === "schedule")) return false;
  }
 
  return true;
}
 
router.get("/api/reports/list", (req, res) => {
  const role = (req.headers["x-user-role"] as string) || "Guest";
  
  const filteredStudios: Record<string, any> = {};
  
  Object.entries(SMRITI_STUDIOS).forEach(([studioKey, studioData]) => {
    const visibleReports = studioData.reports.filter(rpt => 
      checkSmritiReportPermissions(req, "view", rpt.id)
    );
    
    filteredStudios[studioKey] = {
      name: studioData.name,
      description: studioData.description,
      icon: studioData.icon,
      reports: visibleReports
    };
  });
  
  res.json({
    studios: filteredStudios,
    role,
    policyEnforcement: "SMRITI Rule 10 Non-Repudiation Schema Active"
  });
});
 
router.post("/api/reports/schedule", (req, res) => {
  const role = (req.headers["x-user-role"] as string) || "Guest";
  const { reportId, reportName, frequency, deliveryFormat, recipientEmail, cronExpression } = req.body;
  
  if (!reportId || !recipientEmail || !frequency) {
    return res.status(400).json({ error: "Missing required scheduling fields: reportId, recipientEmail, and frequency are mandatory." });
  }
 
  if (!checkSmritiReportPermissions(req, "schedule", reportId)) {
    const role = (req.headers["x-user-role"] as string) || "Guest";
    return res.status(403).json({ error: `Rule 10 Enforced: Role '${role}' is not authorized to register automated email cron schedules.` });
  }
 
  const newSchedule = {
    id: "SCH-" + Date.now(),
    reportId,
    reportName: reportName || "Standard Report",
    frequency,
    deliveryFormat: deliveryFormat || "PDF",
    recipientEmail,
    cronExpression: cronExpression || "0 8 * * *",
    isActive: true,
    createdAt: new Date().toISOString()
  };
 
  reportSchedules.push(newSchedule);
  saveDb();
 
  res.json({ success: true, schedule: newSchedule });
});
 
router.get("/api/reports/schedules", (req, res) => {
  res.json(reportSchedules);
});
 
router.delete("/api/reports/schedule/:id", (req, res) => {
  const { id } = req.params;
  if (!hasPermission(req, "settings.manage")) {
    return res.status(403).json({ error: "Access Denied: You do not have permission to delete report schedules." });
  }
 
  const index = reportSchedules.findIndex(s => s.id === id);
  if (index === -1) {
    return res.status(404).json({ error: "Schedule not found." });
  }
 
  const removed = reportSchedules.splice(index, 1);
  saveDb();
 
  res.json({ success: true, removedSchedule: removed[0] });
});
 
export default router;
