import express from "express";
import { container } from "../bootstrap/di.js";
import { Product } from "../types.js";
 
const router = express.Router();
 
router.get("/api/pos/products", async (req, res) => {
  try {
    const list = await container.products.getAll();
    res.json(list);
  } catch (error) {
    console.error("[SMRITI] Failed to fetch products:", error);
    res.status(500).json({ error: "Internal Server Error retrieving products" });
  }
});
 
router.post("/api/pos/products", async (req, res) => {
  const { name, code, price, stock, category, barcode, color, size, mrp, gstPercentage, styleCode, costPrice, sku } = req.body;
  
  if (!name || !code || !barcode) {
    return res.status(400).json({ error: "Item Name, SKU Code, and Barcode are required." });
  }
 
  try {
    // Check duplicates using repository layer
    const allProds = await container.products.getAll();
    const duplicate = allProds.find(p => p.code === code || p.barcode === barcode);
    if (duplicate) {
      if (duplicate.code === code) {
        return res.status(400).json({ error: `SKU Code ${code} is already registered in SMRITI.` });
      }
      if (duplicate.barcode === barcode) {
        return res.status(400).json({ error: `Barcode ${barcode} is already registered in SMRITI.` });
      }
    }
 
    const pPrice = parseFloat(price) || 0;
    const pStock = parseInt(stock) || 0;
    const pGst = parseFloat(gstPercentage) || 18;
    const pMrp = parseFloat(mrp) || pPrice;
    const pCostPrice = parseFloat(costPrice) || Math.round(pPrice * 0.6);
    const newId = `p-${Date.now()}`;
 
    const created = await container.products.create({
      id: newId,
      name,
      code,
      sku: sku || code,
      price: pPrice,
      costPrice: pCostPrice,
      stock: pStock,
      category: category || "General",
      barcode,
      color: color || undefined,
      size: size || undefined,
      mrp: pMrp,
      gstPercentage: pGst,
      styleCode: styleCode || code
    });
 
    // Audit Logger
    await container.audits.log({
      operator: "Manager Dev",
      actionType: "INSERT",
      tableName: "products",
      recordId: newId,
      oldValue: JSON.stringify("-"),
      newValue: JSON.stringify(`Stock: ${pStock}`),
      reason: `Quick Created SKU: ${code} - ${name}`
    });
 
    res.json(created);
  } catch (error) {
    console.error("[SMRITI] Failed to create product:", error);
    res.status(500).json({ error: "Internal Server Error creating product" });
  }
});
 
router.put("/api/pos/products/:id", async (req, res) => {
  const { id } = req.params;
  const { name, price, stock, category, barcode, color, size, mrp, gstPercentage, styleCode, costPrice, sku } = req.body;
 
  try {
    const oldProduct = await container.products.getById(id);
    if (!oldProduct) {
      return res.status(404).json({ error: "Item not found" });
    }
 
    const updatePayload: Partial<Product> = {};
    if (name !== undefined) updatePayload.name = name;
    if (price !== undefined) updatePayload.price = parseFloat(price) || 0;
    if (stock !== undefined) updatePayload.stock = parseInt(stock) || 0;
    if (category !== undefined) updatePayload.category = category;
    if (barcode !== undefined) updatePayload.barcode = barcode;
    if (color !== undefined) updatePayload.color = color || undefined;
    if (size !== undefined) updatePayload.size = size || undefined;
    if (mrp !== undefined) updatePayload.mrp = parseFloat(mrp) || undefined;
    if (gstPercentage !== undefined) updatePayload.gstPercentage = parseFloat(gstPercentage) || undefined;
    if (styleCode !== undefined) updatePayload.styleCode = styleCode || undefined;
    if (costPrice !== undefined) updatePayload.costPrice = parseFloat(costPrice) || undefined;
    if (sku !== undefined) updatePayload.sku = sku || undefined;
 
    const updated = await container.products.update(id, updatePayload);
 
    // Audit Logger
    await container.audits.log({
      operator: "Manager Dev",
      actionType: "UPDATE",
      tableName: "products",
      recordId: id,
      oldValue: JSON.stringify({ price: oldProduct.price, costPrice: oldProduct.costPrice, stock: oldProduct.stock, mrp: oldProduct.mrp }),
      newValue: JSON.stringify({ price: updated.price, costPrice: updated.costPrice, stock: updated.stock, mrp: updated.mrp }),
      reason: `Updated Master SKU: ${oldProduct.code} (${updated.name})`
    });
 
    res.json(updated);
  } catch (error) {
    console.error("[SMRITI] Failed to update product:", error);
    res.status(500).json({ error: "Internal Server Error updating product" });
  }
});
 
router.post("/api/products/:id/barcodes", async (req, res) => {
  const { id } = req.params;
  const { value } = req.body;
  if (!value) return res.status(400).json({ error: "Barcode value required" });
 
  try {
    const product = await container.products.getById(id);
    if (!product) return res.status(404).json({ error: "Product not found" });
 
    // Check global uniqueness
    const allProds = await container.products.getAll();
    const exists = allProds.some(p => p.barcode === value || (p.secondaryBarcodes && p.secondaryBarcodes.includes(value)));
    if (exists) return res.status(400).json({ error: "Barcode alias already exists globally" });
 
    const updated = await container.products.addBarcode(id, value);
    res.json(updated);
  } catch (error) {
    console.error("[SMRITI] Failed to add secondary barcode:", error);
    res.status(500).json({ error: "Internal Server Error adding barcode" });
  }
});
 
router.delete("/api/products/:id/barcodes/:value", async (req, res) => {
  const { id, value } = req.params;
 
  try {
    const product = await container.products.getById(id);
    if (!product) return res.status(404).json({ error: "Product not found" });
 
    const updated = await container.products.deleteBarcode(id, value);
    res.json(updated);
  } catch (error) {
    console.error("[SMRITI] Failed to delete secondary barcode:", error);
    res.status(500).json({ error: "Internal Server Error deleting barcode" });
  }
});
 
router.delete("/api/pos/products/:id", async (req, res) => {
  const { id } = req.params;
 
  try {
    const product = await container.products.getById(id);
    if (!product) {
      return res.status(404).json({ error: "Item not found" });
    }
 
    await container.products.delete(id);
 
    // Audit Logger
    await container.audits.log({
      operator: "Manager Dev",
      actionType: "DELETE",
      tableName: "products",
      recordId: id,
      oldValue: JSON.stringify(`Stock was: ${product.stock}`),
      newValue: JSON.stringify("Deleted"),
      reason: `Permanently deleted SKU: ${product.code} - ${product.name}`
    });
 
    res.json({ success: true, message: `SKU ${product.code} deleted successfully.` });
  } catch (error) {
    console.error("[SMRITI] Failed to delete product:", error);
    res.status(500).json({ error: "Internal Server Error deleting product" });
  }
});
 
export default router;
